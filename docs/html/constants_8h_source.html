15 // Function templates for safely dealing with pointer referenced items.
16 // The functions defined by these templates may be called using a normal
17 // function call syntax. The compiler will create a function that replaces T
18 // with the type of the calling parameter.
19 //=============================================================================
20 // Safely release pointer referenced item
21 template <typename T>
22 inline void safeRelease(T& ptr)
23 {
24  if (ptr)
25  {
26  ptr->Release();
27  ptr = NULL;
28  }
29 }
30 #define SAFE_RELEASE safeRelease // for backward compatiblility
31 
32 // Safely delete pointer referenced item
33 template <typename T>
34 inline void safeDelete(T& ptr)
35 {
36  if (ptr)
37  {
38  delete ptr;
39  ptr = NULL;
40  }
41 }
42 #define SAFE_DELETE safeDelete // for backward compatiblility
43 
44 // Safely delete pointer referenced array
45 template <typename T>
46 inline void safeDeleteArray(T& ptr)
47 {
48  if (ptr)
49  {
50  delete[] ptr;
51  ptr = NULL;
52  }
53 }
54 #define SAFE_DELETE_ARRAY safeDeleteArray // for backward compatiblility
